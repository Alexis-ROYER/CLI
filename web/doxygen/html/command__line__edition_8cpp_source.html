57 CmdLineEdition& CmdLineEdition::operator=(const CmdLineEdition& CLI_CmdLine)
58 {
59  m_tkLeft.Reset();
60  m_tkRight.Reset();
61 
62  for (tk::Queue<KEY>::Iterator it1 = CLI_CmdLine.m_tkLeft.GetIterator(); CLI_CmdLine.m_tkLeft.IsValid(it1); CLI_CmdLine.m_tkLeft.MoveNext(it1))
63  {
64  m_tkLeft.AddTail(CLI_CmdLine.m_tkLeft.GetAt(it1));
65  }
66  for (tk::Queue<KEY>::Iterator it2 = CLI_CmdLine.m_tkRight.GetIterator(); CLI_CmdLine.m_tkRight.IsValid(it2); CLI_CmdLine.m_tkRight.MoveNext(it2))
67  {
68  m_tkRight.AddTail(CLI_CmdLine.m_tkRight.GetAt(it2));
69  }
70  m_bInsertMode = CLI_CmdLine.m_bInsertMode;
71 
72  return *this;
73 }

75 void CmdLineEdition::Set(const tk::String& TK_Left, const tk::String& TK_Right)
76 {
77  m_tkLeft.Reset();
78  m_tkRight.Reset();
79 
80  // Simply set both parts of the command line.
81  StringDecoder cli_Left((const char*) TK_Left);
82  for ( KEY e_Left = cli_Left.GetKey();
83  (e_Left != NULL_KEY) && (m_tkLeft.GetCount() < MAX_CMD_LINE_LENGTH);
84  e_Left = cli_Left.GetKey())
85  {
86  m_tkLeft.AddTail(e_Left);
87  }
88 
89  StringDecoder cli_Right((const char*) TK_Right);
90  for ( KEY e_Right = cli_Right.GetKey();
91  (e_Right != NULL_KEY) && (m_tkLeft.GetCount() + m_tkRight.GetCount() < MAX_CMD_LINE_LENGTH);
92  e_Right = cli_Right.GetKey())
93  {
94  m_tkRight.AddTail(e_Right);
95  }
96 }

136 void CmdLineEdition::Put(const OutputDevice& CLI_OutputDevice, const tk::Queue<KEY>& TK_Keys)
137 {
138  for ( tk::Queue<KEY>::Iterator it1 = TK_Keys.GetIterator();
139  TK_Keys.IsValid(it1) && (m_tkLeft.GetCount() + m_tkRight.GetCount() + ((m_bInsertMode || m_tkRight.IsEmpty()) ? 1 : 0) <= MAX_CMD_LINE_LENGTH);
140  TK_Keys.MoveNext(it1))
141  {
142  // First of all, append the left part of the command line.
143  m_tkLeft.AddTail(TK_Keys.GetAt(it1));
144 
145  // Print out the characters.
146  CLI_OutputDevice << TK_Keys.GetAt(it1);
147 
148  if (! m_bInsertMode)
149  {
150  // Remove the first characters of the right part of the command line.
151  if (! m_tkRight.IsEmpty())
152  {
153  m_tkRight.RemoveHead();
154  }
155  }
156  }
157  if (m_bInsertMode)
158  {
159  // Refresh the right part of the line.
160  for ( tk::Queue<KEY>::Iterator it2 = m_tkRight.GetIterator();
161  m_tkRight.IsValid(it2);
162  m_tkRight.MoveNext(it2))
163  {
164  CLI_OutputDevice << m_tkRight.GetAt(it2);
165  }
166 
167  // Move the cursor to the left.
168  for (unsigned int ui = m_tkRight.GetCount(); ui > 0; ui --)
169  {
170  CLI_OutputDevice.PutString("\b");
171  }
172  }
173 }

181 void CmdLineEdition::Delete(const OutputDevice& CLI_OutputDevice, const int I_Count)
182 {
183  if (I_Count > 0)
184  {
185  // Delete forward
186 
187  // Find out the pattern to keep, reduce the right part of the command line.
188  unsigned int ui_CharCount = 0;
189  for (ui_CharCount = 0; (ui_CharCount < (unsigned int) I_Count) && (! m_tkRight.IsEmpty()); ui_CharCount ++)
190  {
191  m_tkRight.RemoveHead();
192  }
193  // Print over the kept right part of the command line.
194  CLI_OutputDevice << GetRight();
195  // Blank the useless characters at the end of the line.
196  for (unsigned int ui_Blank = ui_CharCount; ui_Blank > 0; ui_Blank --)
197  {
198  CLI_OutputDevice << " ";
199  }
200  // Move back the cursor.
201  for (unsigned int ui_Back = m_tkRight.GetCount() + ui_CharCount; ui_Back > 0; ui_Back --)
202  {
203  CLI_OutputDevice << "\b";
204  }
205  }
206  else if (I_Count < 0)
207  {
208  // Delete backward.
209 
210  // Find out the pattern to keep, reduce the left part of the commane line.
211  unsigned int ui_CharCount = 0;
212  for (ui_CharCount = 0; (ui_CharCount < (unsigned int) -I_Count) && (! m_tkLeft.IsEmpty()); ui_CharCount ++)
213  {
214  m_tkLeft.RemoveTail();
215  }
216  // Move back the cursor.
217  for (unsigned int ui_Back1 = ui_CharCount; ui_Back1 > 0; ui_Back1 --)
218  {
219  CLI_OutputDevice << "\b";
220  }
221  // Print over the right part of the command line.
222  CLI_OutputDevice << GetRight();
223  // Blank the useless characters at the end of the line.
224  for (unsigned int ui_Blank = ui_CharCount; ui_Blank > 0; ui_Blank --)
225  {
226  CLI_OutputDevice << " ";
227  }
228  // Move back the cursor.
229  for (unsigned int ui_Back2 = m_tkRight.GetCount() + ui_CharCount; ui_Back2 > 0; ui_Back2 --)
230  {
231  CLI_OutputDevice << "\b";
232  }
233  }
234 }

246 void CmdLineEdition::MoveCursor(const OutputDevice& CLI_OutputDevice, const int I_Count)
247 {
248  if (I_Count > 0)
249  {
250  // Move forward
251  for (unsigned int ui_Forward = 0; (ui_Forward < (unsigned int) I_Count) && (! m_tkRight.IsEmpty()); ui_Forward ++)
252  {
253  const KEY e_Key = m_tkRight.RemoveHead();
254  m_tkLeft.AddTail(e_Key);
255  CLI_OutputDevice << e_Key;
256  }
257  }
258  else if (I_Count < 0)
259  {
260  // Move backward.
261  for (unsigned int ui_Backward = 0; (ui_Backward < (unsigned int) -I_Count) && (! m_tkLeft.IsEmpty()); ui_Backward ++)
262  {
263  const KEY e_Key = m_tkLeft.RemoveTail();
264  m_tkRight.AddHead(e_Key);
265  CLI_OutputDevice.PutString("\b");
266  }
267  }

310 const tk::String CmdLineEdition::GetNextWord(void) const
311 {
312  StringDevice cli_Word(MAX_CMD_LINE_LENGTH * 3, false); // Special characters may be encoded with 3 bytes max in utf-8
313 
314  // Skip blank characters.
315  tk::Queue<KEY>::Iterator it = m_tkRight.GetIterator();
316  for ( ; m_tkRight.IsValid(it) && (m_tkRight.GetAt(it) == SPACE); m_tkRight.MoveNext(it))
317  {
318  cli_Word << m_tkRight.GetAt(it);
319  }
320 
321  // Store characters until the next blank character.
322  for ( ; m_tkRight.IsValid(it) && (m_tkRight.GetAt(it) != SPACE); m_tkRight.MoveNext(it))
323  {
324  cli_Word << m_tkRight.GetAt(it);
325  }
326 
327  // Return the next word.
