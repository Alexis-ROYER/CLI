58 void Menu::SetCli(Cli& CLI_Cli)
59 {
60  SyntaxNode::SetCli(CLI_Cli);
61  if (m_pcliHelp == NULL)
62  {
63  Help cli_Help(Help()
64  .AddHelp(Help::LANG_EN, "Get help")
65  .AddHelp(Help::LANG_FR, "Obtenir de l'aide"));
66  m_pcliHelp = dynamic_cast<Keyword*>(& AddElement(new Keyword("help", cli_Help)));
67  m_pcliHelp->AddElement(new Endl(cli_Help));
68  }
69  if (m_pcliExit == NULL)
70  {
71  Help cli_Help(Help()
72  .AddHelp(Help::LANG_EN, Help::Concat("Exit menu '", GetKeyword(), "'"))
73  .AddHelp(Help::LANG_FR, Help::Concat("Sortir du menu '", GetKeyword(), "'")));
74  m_pcliExit = dynamic_cast<Keyword*>(& AddElement(new Keyword("exit", cli_Help)));
75  m_pcliExit->AddElement(new Endl(cli_Help));
76  }
77  if (m_pcliQuit == NULL)
78  {
79  Help cli_Help(Help()
80  .AddHelp(Help::LANG_EN, "Quit")
81  .AddHelp(Help::LANG_FR, "Quitter"));
82  m_pcliQuit = dynamic_cast<Keyword*>(& AddElement(new Keyword("quit", cli_Help)));
83  m_pcliQuit->AddElement(new Endl(cli_Help));
84  }
85  if (m_pcliPwm == NULL)
86  {
87  Help cli_Help(Help()
88  .AddHelp(Help::LANG_EN, "Print Working Menu")
89  .AddHelp(Help::LANG_FR, "Affichage du menu courant"));
90  m_pcliPwm = dynamic_cast<Keyword*>(& AddElement(new Keyword("pwm", cli_Help)));
91  m_pcliPwm->AddElement(new Endl(cli_Help));
92  }
93  if (m_pcliCls == NULL)
94  {
95  Help cli_Help(Help()
96  .AddHelp(Help::LANG_EN, "Clean screen")
97  .AddHelp(Help::LANG_FR, "Nettoyage de l'écran"));
98  m_pcliCls = dynamic_cast<Keyword*>(& AddElement(new Keyword("cls", cli_Help)));
99  m_pcliCls->AddElement(new Endl(cli_Help));
100  }
101 }

103 const bool Menu::ExecuteReserved(const CommandLine& CLI_CommandLine) const
104 {
105  CommandLineIterator it(CLI_CommandLine);
106 
107  if (! it.StepIt()) { return false; }
108  else if (it == *m_pcliHelp)
109  {
110  if (! it.StepIt()) { return false; }
111  if (dynamic_cast<const Endl*>(*it) != NULL)
112  {
113  GetShell().DisplayHelp();
114  return true;
115  }
116  }
117  else if (it == *m_pcliExit)
118  {
119  if (! it.StepIt()) { return false; }
120  if (dynamic_cast<const Endl*>(*it) != NULL)
121  {
122  GetShell().ExitMenu(false);
123  return true;
124  }
125  }
126  else if (it == *m_pcliQuit)
127  {
128  if (! it.StepIt()) { return false; }
129  if (dynamic_cast<const Endl*>(*it) != NULL)
130  {
131  GetShell().Quit();
132  return true;
133  }
134  }
135  else if (it == *m_pcliPwm)
136  {
137  if (! it.StepIt()) { return false; }
138  if (dynamic_cast<const Endl*>(*it) != NULL)
139  {
140  GetShell().PrintWorkingMenu();
141  return true;
142  }
143  }
144  else if (it == *m_pcliCls)
145  {
146  if (! it.StepIt()) { return false; }
147  if (dynamic_cast<const Endl*>(*it) != NULL)
148  {
149  GetShell().CleanScreen(false);
150  return true;
151  }
152  }
153 
154  return false;
155 }
