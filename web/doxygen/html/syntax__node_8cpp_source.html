60 Element& SyntaxNode::AddElement(Element* const PCLI_Element)
61 {
62  CLI_ASSERT(PCLI_Element != NULL);
63 
64  for ( Element::List::Iterator it = m_cliElements.GetIterator();
65  m_cliElements.IsValid(it);
66  m_cliElements.MoveNext(it))
67  {
68  if (const Element* const pcli_Element = m_cliElements.GetAt(it))
69  {
70  if (pcli_Element->GetKeyword() == PCLI_Element->GetKeyword())
71  {
72  if (PCLI_Element != pcli_Element)
73  {
79  delete PCLI_Element;

81  }
82  else

125 const bool SyntaxNode::FindElements(
126  Element::List& CLI_ExactList,
127  Element::List& CLI_NearList,
128  const char* const STR_Keyword
129  ) const
130 {
131  // For each child...
132  for ( Element::List::Iterator it = m_cliElements.GetIterator();
133  m_cliElements.IsValid(it);
134  m_cliElements.MoveNext(it))
135  {
136  if (const Element* const pcli_Element = m_cliElements.GetAt(it))
137  {
138  if (false) {}
139  else if (const SyntaxTag* const pcli_Tag = dynamic_cast<const SyntaxTag*>(pcli_Element))
140  {
141  // Propagate call over child non hollow tag.
142  if (! pcli_Tag->GetbHollow())
143  {
144  if (! pcli_Tag->FindElements(CLI_ExactList, CLI_NearList, STR_Keyword))
145  {
146  return false;
147  }
148  }
149  }
150  else if (const SyntaxRef* const pcli_Ref = dynamic_cast<const SyntaxRef*>(pcli_Element))
151  {
152  // Propagate call over referenced tag.
153  if (! pcli_Ref->GetTag().FindElements(CLI_ExactList, CLI_NearList, STR_Keyword))
154  {
155  return false;
156  }
157  }
158  else if (STR_Keyword == NULL)
159  {
160  // No keyword begun.
161  // Retrieve all sub-elements.
162  if (! CLI_NearList.AddTail(pcli_Element))
163  {
164  GetTraces().Trace(INTERNAL_ERROR) << "SyntaxNode::FindElements(): Not enough space in CLI_ExactList." << endl;
165  }
166  }
167  else
168  {
169  // A beginning of word has been given.
170  const tk::String str_Keyword(MAX_WORD_LENGTH, STR_Keyword);
171 
172  if (const Param* const pcli_Param = dynamic_cast<const Param*>(pcli_Element))
173  {
174  // If the child element is a parameter, check SetstrValue() works for it.
175  if (str_Keyword != "\n")
176  {
177  if (pcli_Param->SetstrValue(str_Keyword))
178  {
179  if (! CLI_NearList.AddTail(pcli_Param))
180  {
181  GetTraces().Trace(INTERNAL_ERROR) << "SyntaxNode::FindElements(): Not enough space in CLI_ExactList." << endl;
182  }
183  if (! CLI_ExactList.AddTail(pcli_Param))
184  {
185  GetTraces().Trace(INTERNAL_ERROR) << "SyntaxNode::FindElements(): Not enough space in CLI_ExactList." << endl;
186  }
187  }
188  }
189  }
190  else if (pcli_Element->GetKeyword().SubString(0, str_Keyword.GetLength()) == str_Keyword)
191  {
192  // Check the beginning of the word for other elements.
193  if (! CLI_NearList.AddTail(pcli_Element))
194  {
195  GetTraces().Trace(INTERNAL_ERROR) << "SyntaxNode::FindElements(): Not enough space in CLI_ExactList." << endl;
196  }
197  if (str_Keyword.GetLength() == pcli_Element->GetKeyword().GetLength())
198  {
199  if (! CLI_ExactList.AddTail(pcli_Element))
200  {
201  GetTraces().Trace(INTERNAL_ERROR) << "SyntaxNode::FindElements(): Not enough space in CLI_ExactList." << endl;
202  }
203  }
204  }
205  }
206  }
207  }
208 
209  return true;
210 }
