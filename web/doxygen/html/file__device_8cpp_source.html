72 const bool InputFileDevice::OpenDevice(void)
73 {
74  CLI_ASSERT(& m_cliOutput != this);
75  if (! m_cliOutput.OpenUp(__CALL_INFO__))
76  {
77  m_cliLastError = m_cliOutput.GetLastError();
78  return false;
79  }
80 
81  if (m_pfFile == NULL)
82  {
83  m_pfFile = fopen(m_strFileName, "r");
84  if (m_pfFile == NULL)
85  {
86  m_cliLastError
87  .SetString(ResourceString::LANG_EN, ResourceString::Concat("Cannot open input file '", m_strFileName, "'"))
88  .SetString(ResourceString::LANG_FR, ResourceString::Concat("Impossible d'ouvrir le fichier d'entrée '", m_strFileName, "'"));
89  return false;
90  }
91  }
92 
93  return true;

121 const KEY InputFileDevice::GetKey(void) const
122 {
123  if (m_pfFile != NULL)
124  {
125  // While there are still characters to read.
126  while ((! m_tkInputBuffer.IsEmpty()) || (! feof(m_pfFile)))
127  {
128  // Check the input buffer
129  if (m_tkInputBuffer.IsEmpty())
130  {
131  if ((m_pfFile != NULL) && (! feof(m_pfFile)))
132  {
133  char str_Buffer[MAX_INPUT_BUFFER_LEN];
134  size_t ui_Bytes = fread(str_Buffer, sizeof(char), sizeof(str_Buffer), m_pfFile);
135  if (ui_Bytes > 0)
136  {
137  for (size_t ui=0; (ui<ui_Bytes) && (ui<sizeof(str_Buffer)); ui++)
138  {
139  m_tkInputBuffer.AddTail(str_Buffer[ui]);
140  }
141  }
142  else
143  {
144  m_cliLastError
145  .SetString(ResourceString::LANG_EN, ResourceString::Concat("Error while reading input file '", m_strFileName, "'"))
146  .SetString(ResourceString::LANG_FR, ResourceString::Concat("Erreur de lecture du fichier d'entrée '", m_strFileName, "'"));
147  return NULL_KEY;
148  }
149  }
150  }
151 
152  // Read the next character.
153  if (! m_tkInputBuffer.IsEmpty())
154  {
155  char c_Char = m_tkInputBuffer.RemoveHead();
156 
157  // Next line/column management & special character management.
158  switch (c_Char)
159  {
160  case '\n':
161  m_iCurrentLine = m_iNextLine; m_iNextLine ++;
162  m_iCurrentColumn = m_iNextColumn; m_iNextColumn = 1;
163  break;
164  case '?':
165  if ((m_iNextLine == m_iCurrentLine) && (m_iNextColumn == m_iCurrentColumn) && (! m_bEnableSpecialCharacters))
166  {
167  // Income of the effective '?' character, after it has been escaped (see below).
168  // This time, step over.
169  m_iNextColumn ++;
170  }
171  else if (! m_bEnableSpecialCharacters)
172  {
173  // Escape the '?' character. Post it back. Do not step over yet.
174  m_iCurrentLine = m_iNextLine;
175  m_iCurrentColumn = m_iNextColumn;
176  c_Char = '\\';
177  m_tkInputBuffer.AddHead('?');
178  }
179  else
180  {
181  // Just step over.
182  m_iCurrentLine = m_iNextLine;
183  m_iCurrentColumn = m_iNextColumn ++;
184  }
185  break;
186  case '\t':
187  if (! m_bEnableSpecialCharacters)
188  {
189  // Change tabs into spaces.
190  c_Char = ' ';
191  }
192  // Step over.
193  m_iCurrentLine = m_iNextLine;
194  m_iCurrentColumn = m_iNextColumn ++;
195  default:
196  // Just step over.
197  m_iCurrentLine = m_iNextLine;
198  m_iCurrentColumn = m_iNextColumn ++;
199  break;
200  }
201 
202  // Character analysis.
203  const KEY e_Key = Char2Key(c_Char);
204  if (e_Key == NULL_KEY)
205  {
206  m_cliLastError
207  .SetString(ResourceString::LANG_EN, ResourceString::Concat("Error while reading input file '", m_strFileName, "'"))
208  .SetString(ResourceString::LANG_FR, ResourceString::Concat("Erreur de lecture du fichier d'entrée '", m_strFileName, "'"));
209  }
210  else if (e_Key == FEED_MORE)
211  {
212  // Let's move on the next character
213  }
214  else
215  {
216  return e_Key;
217  }
218  }
219  }
220  }
221 
222  return NULL_KEY;
223 }

225 const ResourceString InputFileDevice::GetLocation(void) const
226 {
227  ResourceString cli_Location;
228 
229  char str_Location[1024];
230  memset(str_Location, '\0', sizeof(str_Location));
231  const int i_Res = snprintf(str_Location, sizeof(str_Location), "%s:%d", (const char* const) m_strFileName, GetCurrentLine());
232  CheckSnprintfResult(str_Location, sizeof(str_Location), i_Res);
233  for (int i_Lang = 0; i_Lang < ResourceString::LANG_COUNT; i_Lang ++)
234  {
235  cli_Location.SetString((ResourceString::LANG) i_Lang, str_Location);
236  }
237  return cli_Location;
238 }
