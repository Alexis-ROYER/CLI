142 const bool ConfigMenu::ExecuteReserved(const CommandLine& CLI_CmdLine) const
143 {
144  CommandLineIterator it(CLI_CmdLine);
145 
146  if (! it.StepIt()) { return false; }
147  else if (it == GetShowNode())
148  {
149  if (! it.StepIt()) { return false; }
150  else if (it == GetShowVersionNode())
151  {
152  if (! it.StepIt()) { return false; }
153  if (dynamic_cast<const Endl*>(*it))
154  {
155  ShowVersion();
156  return true;
157  }
158  }
159  else if (it == GetShowAuthorNode())
160  {
161  if (! it.StepIt()) { return false; }
162  if (dynamic_cast<const Endl*>(*it))
163  {
164  ShowAuthor();
165  return true;
166  }
167  }
168  else if (it == GetShowLicenseNode())
169  {
170  if (! it.StepIt()) { return false; }
171  if (dynamic_cast<const Endl*>(*it))
172  {
173  ShowLicense();
174  return true;
175  }
176  }
177  }
178  else if (it == GetEchoNode())
179  {
180  if (! it.StepIt()) { return false; }
181  else if (it == GetEchoOnNode())
182  {
183  if (! it.StepIt()) { return false; }
184  if (dynamic_cast<const Endl*>(*it))
185  {
186  EchoOn();
187  return true;
188  }
189  }
190  else if (it == GetEchoOffNode())
191  {
192  if (! it.StepIt()) { return false; }
193  if (dynamic_cast<const Endl*>(*it))
194  {
195  EchoOff();
196  return true;
197  }
198  }
199  }
200  else if (it == GetBeepNode())
201  {
202  if (! it.StepIt()) { return false; }
203  else if (it == GetBeepOnNode())
204  {
205  if (! it.StepIt()) { return false; }
206  if (dynamic_cast<const Endl*>(*it))
207  {
208  BeepOn();
209  return true;
210  }
211  }
212  else if (it == GetBeepOffNode())
213  {
214  if (! it.StepIt()) { return false; }
215  if (dynamic_cast<const Endl*>(*it))
216  {
217  BeepOff();
218  return true;
219  }
220  }
221  }
222  else if (it == GetLangNode())
223  {
224  if (! it.StepIt()) { return false; }
225  else if (it == GetEnglishLangNode())
226  {
227  if (! it.StepIt()) { return false; }
228  if (dynamic_cast<const Endl*>(*it))
229  {
230  SetLang(ResourceString::LANG_EN);
231  return true;
232  }
233  }
234  else if (it == GetFrenchLangNode())
235  {
236  if (! it.StepIt()) { return false; }
237  if (dynamic_cast<const Endl*>(*it))
238  {
239  SetLang(ResourceString::LANG_FR);
240  return true;
241  }
242  }
243  }
244  #ifdef _DEBUG
245  else if (it == GetCheckNode())
246  {
247  if (! it.StepIt()) { return false; }
248  else if (it == GetCheckIONode())
249  {
250  if (! it.StepIt()) { return false; }
251  else if (dynamic_cast<const Endl*>(*it))
252  {
253  CheckIODevice();
254  return true;
255  }
256  else if (it == GetOutCharNode())
257  {
258  if (! it.StepIt()) { return false; }
259  if (dynamic_cast<const Endl*>(*it))
260  {
261  CheckOutChar();
262  return true;
263  }
264  }
265  }
266  }
267  #endif
268 
269  return Menu::ExecuteReserved(CLI_CmdLine);
270 }

337 #ifdef _DEBUG
338 void ConfigMenu::CheckIODevice(void) const
339 {
340  class Press {
341  public: static const bool Key(const Shell& CLI_Shell, const char* const STR_Message, const KEY E_Key) {
342  if (E_Key == BREAK)
343  {
344  CLI_Shell.GetStream(OUTPUT_STREAM) << endl;
345  CLI_Shell.GetStream(OUTPUT_STREAM) << "Please follow the instructions, and press one by one the expected characters." << endl;
346  CLI_Shell.GetStream(OUTPUT_STREAM) << endl;
347  }
348 
349  for (KEY e_Key = NULL_KEY; e_Key != E_Key; )
350  {
351  CLI_Shell.GetStream(OUTPUT_STREAM) << STR_Message << endl;
352  e_Key = CLI_Shell.GetInput().GetKey();
353 
354  if ((e_Key == NULL_KEY)
355  || ((e_Key == BREAK) && (E_Key != BREAK)))
356  {
357  // Abort
358  return false;
359  }
360  if ((e_Key == ESCAPE) && (E_Key != ESCAPE) && (E_Key != BREAK))
361  {
362  // Skip character
363  return true;
364  }
365 
366  if (e_Key != E_Key)
367  {
368  CLI_Shell.GetStream(ERROR_STREAM) << "Invalid key " << (int) e_Key << endl;
369  }
370  }
371 
372  if (E_Key == ESCAPE)
373  {
374  CLI_Shell.GetStream(OUTPUT_STREAM) << endl;
375  CLI_Shell.GetStream(OUTPUT_STREAM) << "From now, press ESCAPE to skip a character entry, press CTRL+C to abort the procedure." << endl;
376  CLI_Shell.GetStream(OUTPUT_STREAM) << endl;
377  }
378 
379  return true;
380  }
381  };
382
