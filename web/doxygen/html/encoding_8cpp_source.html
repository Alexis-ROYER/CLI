312 const KEY StringDecoder::GetKey(void)
313 {
314  if (! m_tkString.IsEmpty())
315  {
316  const char c_Char = m_tkString.GetChar(0);
317  m_tkString = m_tkString.SubString(1, m_tkString.GetLength() - 1);
318  const KEY e_Key = Decode(c_Char);
319  if (e_Key == FEED_MORE)
320  {
321  // Recursive call
322  return GetKey();
323  }
324  return e_Key;
325  }
326  return NULL_KEY;
327 }
