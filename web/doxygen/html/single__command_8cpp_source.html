86 const KEY SingleCommand::GetKey(void) const
87 {
88  if ((m_iPosition >= 0) && (m_iPosition < (signed) m_strCommand.GetLength()))
89  {
90  const char c_Char = m_strCommand[m_iPosition];
91  m_iPosition ++;
92  const KEY e_Key = Char2Key(c_Char);
93  if (e_Key == NULL_KEY)
94  {
95  // Do nothing, just ignore.
96  // Recursive call.
97  return GetKey();
98  }
99  else if (e_Key == FEED_MORE)
100  {
101  // Recursive call.
102  return GetKey();
103  }
104  else
105  {
106  return e_Key;
107  }
108  }
109  else if (m_iPosition == (signed) m_strCommand.GetLength())
110  {
111  m_iPosition ++;
112  return ENTER;
113  }
114  else
115  {
116  m_cliLastError
117  .SetString(ResourceString::LANG_EN, "No more characters in command line")
118  .SetString(ResourceString::LANG_FR, "Il n'y a plus de caractères dans la ligne de commande");
119  return NULL_KEY;
120  }
121 }
