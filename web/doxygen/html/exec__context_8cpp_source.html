119 void ExecutionContext::Run(IODevice& CLI_IODevice)
120 {
121  if (BeginExecution(CLI_IODevice))
122  {
123  if (dynamic_cast<NonBlockingIODevice*>(& CLI_IODevice) == NULL)
124  {
125  // Call the main blocking loop.
126  MainBlockingLoop();
127  // Once the main loop is done, terminate the thing.
128  FinishExecution();
129  }
130  }
131  else
132  {
133  FinishExecution();
134  }
135 }

142 const bool ExecutionContext::BeginExecution(IODevice& CLI_IODevice)
143 {
144  GetTraces().Trace(TRACE_EXEC_CTX) << "Execution context " << this << ": Starting execution." << endl;
145 
146  if (m_cliManager.OpenUp(*this, CLI_IODevice))
147  {
148  // Let the actual execution context make its own start ups.
149  if (OnStartExecution())
150  {
151  return true;
152  }
153  }
154 
155  return false;
156 }

158 void ExecutionContext::MainBlockingLoop(void)
159 {
160  // While there are menus in the menu stack, it means we are still waiting for command lines.
161  while (IsRunning())
162  {
163  // Get an input key.
164  const KEY e_Key = m_cliManager.GetInput().GetKey();
165  if (e_Key != NULL_KEY)
166  {
167  // Process the input key.
168  ProcessKey(e_Key);
169  }
170  else
171  {
172  // End of input.
173  FinishExecution();
174  }
175  }
176 }

178 void ExecutionContext::ProcessKey(const KEY E_KeyCode)
179 {
180  const OutputDevice& cli_Trace = GetTraces().Trace(TRACE_EXEC_CTX) << "Execution context " << this << ": ";
181  switch (E_KeyCode)
182  {
183  case ENTER: cli_Trace << "Key " << E_KeyCode << "/\\n"; break;
184  default: cli_Trace << "Key " << E_KeyCode << "/" << (char) E_KeyCode; break;
185  }
186  cli_Trace << " received." << endl;
187 
188  if (IsRunning())
189  {
190  if (E_KeyCode != NULL_KEY)
191  {
192  // Process the input key.
193  OnKey(E_KeyCode);
194  }
195  else
196  {
197  // End of input.
198  FinishExecution();
199  }
200  }

203 const bool ExecutionContext::FinishExecution(void)
204 {
205  GetTraces().Trace(TRACE_EXEC_CTX) << "Execution context " << this << ": Ending execution." << endl;
206 
207  bool b_Res = true;
208 
209  // Let the actual execution context make its own liberations.
210  if (! OnStopExecution())
211  {
212  b_Res = false;
213  }
214 
215  // Do not terminate twice.
216  if (IsRunning())
217  {
218  if (! m_cliManager.CloseDown(*this))
219  {
220  b_Res = false;
221  }
222 
223  // Call registered result interfaces at the very end.
224  tk::Queue<ExecutionResult*> tk_Results(m_tkResults.GetCount());
225  while (! m_tkResults.IsEmpty())
226  {
227  tk_Results.AddTail(m_tkResults.RemoveHead());
228  }
229  while (! tk_Results.IsEmpty())
230  {
231  if (ExecutionResult* const pcli_ExecutionResult = tk_Results.RemoveTail())
232  {
233  GetTraces().Trace(TRACE_EXEC_CTX) << "Execution context " << this << ": Notifying result to interface " << pcli_ExecutionResult << "." << endl;
234  pcli_ExecutionResult->OnResult(*this);
235  }
236  }
237  }
238 
239  return b_Res;
240 }
