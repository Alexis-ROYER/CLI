84 const tk::String SyntaxRef::GetKeyword(void) const
85 {
86  static char str_Buffer[256];
87  const int i_Res = snprintf(str_Buffer, sizeof(str_Buffer), "ref[%p]", this);
88  CheckSnprintfResult(str_Buffer, sizeof(str_Buffer), i_Res);
89 
90  tk::String str_Keyword(MAX_WORD_LENGTH);
91  if (! str_Keyword.Set(str_Buffer))
92  {
93  GetTraces().Trace(INTERNAL_ERROR) << "SyntaxRef::GetKeyword() could not construct the SyntaxRef keyword from string '" << str_Buffer << "'" << endl;
94  }
95  if (m_pcliTag != NULL)
96  {
97  if ((! str_Keyword.Append(" -> "))
98  || (! str_Keyword.Append(m_pcliTag->GetKeyword())))
99  {
100  GetTraces().Trace(INTERNAL_ERROR) << "SyntaxRef::GetKeyword() could not construct the SyntaxRef keyword from strings ' -> ' or m_pcliTag" << endl;
101  }
102  }
103  return str_Keyword;
104 }
