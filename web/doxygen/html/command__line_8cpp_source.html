89 const bool CommandLine::Parse(
90  const Menu& CLI_Menu,
91  const tk::String& STR_Line,
92  const bool B_Execution
93  )
94 {
95  m_pcliMenu = & CLI_Menu;
96 
97  // Reset the error message.
98  m_cliError = ResourceString();
99 
100  // Split the line into words.
101  const Element* pcli_Node = & CLI_Menu;
102  tk::Queue<tk::String> vstr_Words(MAX_CMD_LINE_WORD_COUNT);
103  int i_LastWordPosition = -1;
104  if (! Split(STR_Line, vstr_Words, i_LastWordPosition))
105  {
106  GetTraces().Trace(TRACE_CMD_LINE) << "Could not split '" << STR_Line << "' (" << m_cliError.GetString(ResourceString::LANG_EN) << ")" << endl;
107  return false;
108  }
109 
110  // Determine whether the last word should be parsed.
111  int i_WordCount = vstr_Words.GetCount();
112  if ((! B_Execution)
113  && (! vstr_Words.IsEmpty())
114  && (i_LastWordPosition >= 0))
115  {
116  i_WordCount --;
117  if (! m_strLastWord.Set(vstr_Words.GetTail()))
118  {
119  GetTraces().Trace(INTERNAL_ERROR)
120  << "CommandLine::Parse(): "
121  << "Not enough space in m_strLastWord for the word '" << vstr_Words.GetTail() << "'"
122  << endl;
123  }
124  m_bLastWordValid = true;
125  m_iNumBackspacesForCompletion = (STR_Line.GetLength() - i_LastWordPosition);
126  }
127  else
128  {
129  m_strLastWord.Set("");
130  m_bLastWordValid = false;
131  m_iNumBackspacesForCompletion = 0;
132  }
133 
134  // For each word, match the right element.
135  tk::Queue<tk::String>::Iterator it = vstr_Words.GetIterator();
136  for ( int i=0;
137  vstr_Words.IsValid(it) && (i < i_WordCount);
138  i ++)
139  {
140  // Search for elements that match the current word.
141  GetTraces().Trace(TRACE_CMD_LINE) << "Word " << i << " '" << vstr_Words.GetAt(it) << "'" << endl;
142  Element::List cli_ExactList(MAX_CMD_LINE_WORD_COUNT), cli_NearList(MAX_CMD_LINE_WORD_COUNT);
143  if (! pcli_Node->FindElements(cli_ExactList, cli_NearList, vstr_Words.GetAt(it)))
144  {
145  m_cliError
146  .SetString(ResourceString::LANG_EN, "Internal error")
147  .SetString(ResourceString::LANG_FR, "Erreur interne");
148  GetTraces().Trace(TRACE_CMD_LINE) << m_cliError.GetString(ResourceString::LANG_EN) << endl;
149  return false;
150  }
151 
152  if (cli_NearList.GetCount() == 0)
153  {
154  if (vstr_Words.GetAt(it) == "\n")
155  {
156  if (i == 0)
157  {
158  // End of line
159  return true;
160  }
161  else
162  {
163  // Incomplete command
164  m_cliError
165  .SetString(ResourceString::LANG_EN, "Incomplete command")
166  .SetString(ResourceString::LANG_FR, "Commande incomplète");
167  GetTraces().Trace(TRACE_CMD_LINE) << m_cliError.GetString(ResourceString::LANG_EN) << endl;
168  return false;
169  }
170  }
171  else
172  {
173  // No matching element.
174  // Syntax error.
175  m_cliError
176  .SetString(ResourceString::LANG_EN, ResourceString::Concat("Syntax error next to '", vstr_Words.GetAt(it), "'"))
177  .SetString(ResourceString::LANG_FR, ResourceString::Concat("Erreur de syntaxe près de '", vstr_Words.GetAt(it), "'"));
178  GetTraces().Trace(TRACE_CMD_LINE) << m_cliError.GetString(ResourceString::LANG_EN) << endl;
179  return false;
180  }
181  }
182  else if ((cli_ExactList.GetCount() > 1)
183  || ((cli_ExactList.GetCount() == 0) && (cli_NearList.GetCount() > 1)))
184  {
185  // Ambiguous syntax
186  m_cliError
187  .SetString(ResourceString::LANG_EN, ResourceString::Concat("Ambiguous syntax next to '", vstr_Words.GetAt(it), "'"))
188  .SetString(ResourceString::LANG_FR, ResourceString::Concat("Ambiguïté de syntaxe près de '", vstr_Words.GetAt(it), "'"));
189  GetTraces().Trace(TRACE_CMD_LINE) << m_cliError.GetString(ResourceString::LANG_EN) << endl;
190  return false;
191  }
192  else
193  {
194  // Element found.
195  if (! cli_ExactList.IsEmpty())
196  {
197  // Prefer exactly matching elements if any.
198  pcli_Node = cli_ExactList.GetHead();
199  }
200  else if (! cli_NearList.IsEmpty())
201  {
202  // First of the near elements otherwise.
203  pcli_Node = cli_NearList.GetHead();
204  }
205  else
206  {
207  CLI_ASSERT(false);
208  }
209  // Add it to the command line.
210  AddElement(pcli_Node);
211  }
212 
213  vstr_Words.MoveNext(it);
214  }
215 
216  // Check for final endl element.
217  if (B_Execution && (i_WordCount > 0)
218  && (dynamic_cast<const Endl*>(& GetLastElement()) == NULL))
219  {
220  m_cliError
221  .SetString(ResourceString::LANG_EN, "Incomplete command")
222  .SetString(ResourceString::LANG_FR, "Commande incomplète");
223  GetTraces().Trace(TRACE_CMD_LINE) << m_cliError.GetString(ResourceString::LANG_EN) << endl;
224  return false;
225  }
226 
227  if (GetLastWord() != NULL)
228  {
229  GetTraces().Trace(TRACE_CMD_LINE) << "Last word '" << GetLastWord() << "'..." << endl;
230  }
231  return true;
232 }

268 const bool CommandLine::Split(
269  const tk::String& STR_Line,
270  tk::Queue<tk::String>& VSTR_Words,
271  int& I_LastWordPosition
272  )
273 {
274  I_LastWordPosition = -1;
275 
276  class Do { public:
277  static const bool PushWord(
278  const Element& CLI_Element, tk::Queue<tk::String>& VSTR_Words,
279  const int I_Position, tk::String& STR_Word,
280  ResourceString& CLI_Error)
281  {
282  if (VSTR_Words.AddTail(STR_Word))
283  {
284  GetTraces().Trace(TRACE_CMD_LINE_SPLIT) << "Word '" << STR_Word << "' pushed at " << I_Position << endl;
285  STR_Word.Set("");
286  return true;
287  }
288  else
289  {
290  CLI_Error
291  .SetString(ResourceString::LANG_EN, "Too many words in command line")
292  .SetString(ResourceString::LANG_FR, "Trop de mots dans la ligne de commande");
293  GetTraces().Trace(TRACE_CMD_LINE_SPLIT) << CLI_Error.GetString(ResourceString::LANG_EN) << " on '" << STR_Word << "'" << endl;
294  return false;
295  }
296  }
297  };
298 
299  tk::String str_Word(MAX_WORD_LENGTH);
300  bool b_EscapeMode = false;
301  bool b_QuotedWord = false;
302  for (unsigned int i=0; i<STR_Line.GetLength(); i++)
303  {
304  const char c = STR_Line[i];
305 
306  // End of line.
307  if (c == '\n')
308  {
309  if (b_EscapeMode)
310  {
311  m_cliError
312  .SetString(ResourceString::LANG_EN, "Unterminated escape sequence")
313  .SetString(ResourceString::LANG_FR, "Séquence d'échappement incomplète");
314  GetTraces().Trace(TRACE_CMD_LINE_SPLIT) << m_cliError.GetString(ResourceString::LANG_EN) << " on \\n" << endl;
315  return false;
316  }
317  if (b_QuotedWord)
318  {
319  m_cliError
320  .SetString(ResourceString::LANG_EN, "Unterminated quoted string")
321  .SetString(ResourceString::LANG_FR, "Guillemets non fermés");
322  GetTraces().Trace(TRACE_CMD_LINE_SPLIT) << m_cliError.GetString(ResourceString::LANG_EN) << " on \\n" << endl;
323  return false;
324  }
325  if (! str_Word.IsEmpty())
326  {
327  if (! Do::PushWord(*m_pcliMenu, VSTR_Words, i, str_Word, m_cliError)) return false;
328  GetTraces().Trace(TRACE_CMD_LINE_SPLIT) << "LastWordPosition set to -1 on \\n" << endl;
329  I_LastWordPosition = -1;
330  }
331  }
332  // Quote
333  else if ((c == '"') && (! b_EscapeMode))
334  {
335  if (! b_QuotedWord)
336  {
337  // Beginning of quoted string.
338  GetTraces().Trace(TRACE_CMD_LINE_SPLIT) << "Quoted string starting at " << i << endl;
339  b_QuotedWord = true;
340  // Push previous word if any.
341  if (! str_Word.IsEmpty())
342  {
343  if (! Do::PushWord(*m_pcliMenu, VSTR_Words, i, str_Word, m_cliError)) return false;
344  }
345  // Completion management.
346  GetTraces().Trace(TRACE_CMD_LINE_SPLIT) << "LastWordPosition set to " << i << " on \"" << endl;
347  I_LastWordPosition = i;
348  }
349  else
350  {
351  // End of quoted string.
352  GetTraces().Trace(TRACE_CMD_LINE_SPLIT) << "Quoted string ending at " << i << endl;
353  b_QuotedWord = false;
354  // Push the word whatever.
355  if (! Do::PushWord(*m_pcliMenu, VSTR_Words, i, str_Word, m_cliError)) return false;
356  // No completion on quoted strings.
357  GetTraces().Trace(TRACE_CMD_LINE_SPLIT) << "LastWordPosition set to -1 on \"" << endl;
358  I_LastWordPosition = -1;
359  }
360  }
361  // Blank characters.
362  else if (((c == ' ') || (c == '\t')) && (! b_EscapeMode) && (! b_QuotedWord))
363  {
364  if (! str_Word.IsEmpty())
365  {
366  if (! Do::PushWord(*m_pcliMenu, VSTR_Words, i, str_Word, m_cliError)) return false;
367  GetTraces().Trace(TRACE_CMD_LINE_SPLIT) << "LastWordPosition set to -1 on blank character" << endl;
368  I_LastWordPosition = -1;
369  }
370  }
371  // Escape character.
372  else if ((c == '\\') && (! b_EscapeMode) && (i < STR_Line.GetLength() - 1))
373  {
374  GetTraces().Trace(TRACE_CMD_LINE_SPLIT) << "Escape mode starting at " << i << endl;
375  b_EscapeMode = true;
376  if (I_LastWordPosition < 0)
377  {
378  GetTraces().Trace(TRACE_CMD_LINE_SPLIT) << "LastWordPosition set to " << i << " on \\" << endl;
379  I_LastWordPosition = i;
380  }
381  }
382  // Non blank character.
383  else
384  {
385  // Word expansion.
386  if (! str_Word.Append(c))
387  {
388  // Internal error.
389  m_cliError
390  .SetString(ResourceString::LANG_EN, ResourceString::Concat("Too long word '", str_Word.SubString(0, 10), "...'"))
391  .SetString(ResourceString::LANG_FR, ResourceString::Concat("Mot trop long '", str_Word.SubString(0, 10), "...'"));
392  GetTraces().Trace(TRACE_CMD_LINE_SPLIT) << m_cliError.GetString(ResourceString::LANG_EN) << " on " << c << endl;
393  return false;
394  }
395  if (I_LastWordPosition < 0)
396  {
397  GetTraces().Trace(TRACE_CMD_LINE_SPLIT) << "LastWordPosition set to " << i << " on non-blank character" << endl;
398  I_LastWordPosition = i;
399  }
400 
401  // Escape mode reset.
402  if (b_EscapeMode)
403  {
404  GetTraces().Trace(TRACE_CMD_LINE_SPLIT) << "Escape mode ending at " << i << endl;
405  b_EscapeMode = false;
406  }
407  }
408 
409  // End of line management.
410  if (i >= STR_Line.GetLength() - 1)
411  {
412  if (! str_Word.IsEmpty())
413  {
414  if (! Do::PushWord(*m_pcliMenu, VSTR_Words, i, str_Word, m_cliError)) return false;
415  }
416  }
417 
418  if (c == '\n')
419  {
420  if (VSTR_Words.AddTail(tk::String(MAX_WORD_LENGTH, "\n")))
421  {
422  GetTraces().Trace(TRACE_CMD_LINE_SPLIT) << "\\n pushed at " << i << endl;
423  GetTraces().Trace(TRACE_CMD_LINE_SPLIT) << "LastWordPosition set to -1 on \\n" << endl;
424  I_LastWordPosition = -1;
425  return true;
426  }
427  else
428  {
429  m_cliError
430  .SetString(ResourceString::LANG_EN, ResourceString::Concat("Too long word '", str_Word.SubString(0, 10), "...'"))
431  .SetString(ResourceString::LANG_FR, ResourceString::Concat("Mot trop long '", str_Word.SubString(0, 10), "...'"));
432  GetTraces().Trace(TRACE_CMD_LINE_SPLIT) << m_cliError.GetString(ResourceString::LANG_EN) << " on \\n" << endl;
433  return false;
434  }
435  }
436  }
437 
438  if (b_EscapeMode)
439  {
440  m_cliError
441  .SetString(ResourceString::LANG_EN, "Unterminated escape sequence")
442  .SetString(ResourceString::LANG_FR, "Séquence d'échappement incomplète");
443  GetTraces().Trace(TRACE_CMD_LINE_SPLIT) << m_cliError.GetString(ResourceString::LANG_EN) << " on end of line" << endl;
444  return false;
445  }
446  if (b_QuotedWord)
447  {
448  m_cliError
449  .SetString(ResourceString::LANG_EN, "Unterminated quoted string")
450  .SetString(ResourceString::LANG_FR, "Guillemets non fermés");
451  GetTraces().Trace(TRACE_CMD_LINE_SPLIT) << "Unterminated quoted string on end of line" << endl;
452  return false;
453  }
454  return true;
455 }
