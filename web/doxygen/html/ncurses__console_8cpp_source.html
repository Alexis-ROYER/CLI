103 const bool Console::OpenDevice(void)
104 {
105  if (m_pData == NULL)
106  {
107  // Ensure UTF-8 output before configuring ncurses.
108  // See https://stackoverflow.com/questions/9922528/how-to-make-ncurses-display-utf-8-chars-correctly-in-c
109  const char* const str_PreviousLocale = setlocale(LC_CTYPE, "");
110  GetTraces().SafeTrace(CLI_NCURSES_CONSOLE, *this) << "Console::OpenDevice(): setlocale() returned '" << str_PreviousLocale << "'" << endl;
111 
112  // Configure ncurses
113 
114  // Regular ncurses configuration.
115  WINDOW* const p_Window = initscr();
116  raw(); // Use raw() instead of cbreak(), since we also want to trap the control character sequences.
117  noecho();
118  // Additional configuration.
119  scrollok(p_Window, TRUE); // Enable scrolling.
120  keypad(p_Window, TRUE); // Enable arrow keys.
121  set_escdelay(25); // Change the escape delay to 25ms instead of 1000 ms by default.
122  // see http://en.chys.info/2009/09/esdelay-ncurses/
123 
124  m_pData = new NCursesConsole(p_Window);
125 
126  NCursesConsole::m_iDeviceCount ++;
127  }
128 
129  if (m_pData == NULL)
130  {
131  m_cliLastError
132  .SetString(ResourceString::LANG_EN, "ncurses configuration failed")
133  .SetString(ResourceString::LANG_FR, "La configuration de ncurses a échoué");
134  return false;
135  }
136  else
137  {
138  return true;
139  }
140 }

142 const bool Console::CloseDevice(void)
143 {
144  if (NCursesConsole* const pcli_Data = (NCursesConsole*) m_pData)
145  {
146  NCursesConsole::m_iDeviceCount --;
147  if (NCursesConsole::m_iDeviceCount <= 0)
148  {
149  // Restore original escape delay.
150  set_escdelay(ORIGINAL_DELAY);
151  }
152 
153  endwin();
154 
155  delete pcli_Data;
156  m_pData = NULL;
157  }
158 
159  return true;

342 void Console::PutString(const char* const STR_Out) const
343 {
344  char str_Char[2] = { '\0', '\0' };
345 
346  for (const char* pc_Out = STR_Out; (pc_Out != NULL) && (*pc_Out != '\0'); pc_Out++)
347  {
348  str_Char[0] = *pc_Out;
349  // Possible translation of certain character here...
350  // ...
351 
352  // Output characters.
353  addstr(str_Char);
354 
355  // ncurses seems to have refresh troubles with 'more' page displays.
356  // However, it works good with 'less' page displays.
357  // Over one page, refreshing every new line is a workaround that fixes that behaviour.
358  if (*pc_Out == '\n')
359  {
360  if (NCursesConsole* const pcli_Data = (NCursesConsole*) m_pData)
361  {
362  pcli_Data->m_uiLineCount ++;
363  if ((int) pcli_Data->m_uiLineCount >= LINES)
364  {
365  refresh();
366  }
367  }
368  }
369  }
370 
371  // Eventually referesh the screen.
372  refresh();
373 }
