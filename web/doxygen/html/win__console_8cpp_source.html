264 void Console::PutString(const char* const STR_Out) const
265 {
266  if (STR_Out != NULL)
267  {
268  // Windows console works better when unicode string is output.
269  // See https://msdn.microsoft.com/en-us/library/windows/desktop/dd319072(v=vs.85).aspx
270  int i_RequiredWideCharLen = MultiByteToWideChar(
271  CP_UTF8, 0, // CodePage, dwFlags
272  STR_Out, -1, // lpMultiByteStr
273  NULL, 0 // lpWideCharStr
274  );
275  if (i_RequiredWideCharLen > 0) {
276  LPWSTR str_Unicode = new WCHAR[i_RequiredWideCharLen];
277  MultiByteToWideChar(
278  CP_UTF8, 0, // CodePage, dwFlags
279  STR_Out, -1, // lpMultiByteStr
280  str_Unicode, i_RequiredWideCharLen // lpWideCharStr
281  );
282  _cputws(str_Unicode);
283  delete [] str_Unicode;
284  }
285 
286  // Insure the cursor is visible (reposition in the same place).
287  do
288  {
289  HANDLE h_Console = ::GetWinConsoleHandle();
290  CONSOLE_SCREEN_BUFFER_INFO t_ScreenBufferInfo;
291  if (! ::GetConsoleScreenBufferInfo(h_Console, & t_ScreenBufferInfo))
292  {
293  GetTraces().SafeTrace(CLI_WIN_CONSOLE_OUT, *this) << "GetConsoleScreenBufferInfo(line=" << __LINE__ << ") failed (" << (uint64_t) ::GetLastError() << ")" << endl;
294  break;
295  }
296  if (! ::SetConsoleCursorPosition(h_Console, t_ScreenBufferInfo.dwCursorPosition))
297  {
298  GetTraces().SafeTrace(CLI_WIN_CONSOLE_OUT, *this) << "GetConsoleScreenBufferInfo(line=" << __LINE__ << ") failed (" << (uint64_t) ::GetLastError() << ")" << endl;
299  break;
300  }
301  } while(0);
302  }

310 void Console::CleanScreen(void) const
311 {
312  do
313  {
314  // Retrieve the handle of the console (use stdout).
315  HANDLE h_Console = ::GetWinConsoleHandle();
316 
317  // Get the number of character cells in the current buffer.
318  CONSOLE_SCREEN_BUFFER_INFO t_ScreenBufferInfo;
319  if (! ::GetConsoleScreenBufferInfo(h_Console, & t_ScreenBufferInfo))
320  {
321  GetTraces().SafeTrace(CLI_WIN_CONSOLE_OUT, *this) << "GetConsoleScreenBufferInfo(line=" << __LINE__ << ") failed (" << (uint64_t) ::GetLastError() << ")" << endl;
322  break;
323  }
324  const DWORD dw_ConsoleSize = t_ScreenBufferInfo.dwSize.X * t_ScreenBufferInfo.dwSize.Y;
325 
326  // Fill the entire screen with blanks.
327  const COORD t_ZeroCoord = { 0, 0 };
328  DWORD dw_CharsWritten = 0;
329  if (! ::FillConsoleOutputCharacter(h_Console, (TCHAR) ' ', dw_ConsoleSize, t_ZeroCoord, & dw_CharsWritten))
330  {
331  GetTraces().SafeTrace(CLI_WIN_CONSOLE_OUT, *this) << "FillConsoleOutputCharacter(line=" << __LINE__ << ") failed (" << (uint64_t) ::GetLastError() << ")" << endl;
332  break;
333  }
334 
335 // This part of code seems a little bit strange to me,
336 // Whatever, it is advised by MSDN. Nevermind then.
337  // Get the current text attribute.
338  if (! ::GetConsoleScreenBufferInfo(h_Console, & t_ScreenBufferInfo))
339  {
340  GetTraces().SafeTrace(CLI_WIN_CONSOLE_OUT, *this) << "GetConsoleScreenBufferInfo(line=" << __LINE__ << ") failed (" << (uint64_t) ::GetLastError() << ")" << endl;
341  break;
342  }
343 
344  // Now set the buffer's attributes accordingly.
345  if (! ::FillConsoleOutputAttribute(h_Console, t_ScreenBufferInfo.wAttributes, dw_ConsoleSize, t_ZeroCoord, & dw_CharsWritten))
346  {
347  GetTraces().SafeTrace(CLI_WIN_CONSOLE_OUT, *this) << "FillConsoleOutputAttribute(line=" << __LINE__ << ") failed (" << (uint64_t) ::GetLastError() << ")" << endl;
348  break;
349  }
350 // End of strange code
351 
352  // Put the cursor at (0, 0).
353  if (! ::SetConsoleCursorPosition(h_Console, t_ZeroCoord))
354  {
355  GetTraces().SafeTrace(CLI_WIN_CONSOLE_OUT, *this) << "SetConsoleCursorPosition(line=" << __LINE__ << ") failed (" << (uint64_t) ::GetLastError() << ")" << endl;
356  break;
357  }
358 
359  // Succesful return.
360  return;
361  } while(0);
362 
363  // Fallback to default behaviour.
364  IODevice::CleanScreen();
365 }

367 const OutputDevice::ScreenInfo Console::GetScreenInfo(void) const
368 {
369  do
370  {
371  // Retrieve the handle of the console (use stdout).
372  HANDLE h_Console = ::GetWinConsoleHandle();
373 
374  // Get the number of character cells in the current buffer.
375  CONSOLE_SCREEN_BUFFER_INFO t_ScreenBufferInfo;

378  GetTraces().SafeTrace(CLI_WIN_CONSOLE_OUT, *this) << "GetConsoleScreenBufferInfo(line=" << __LINE__ << ") failed (" << (uint64_t) ::GetLastError() << ")" << endl;
379  break;
380  }
381 
382  DWORD dw_Mode = 0;
383  if (! ::GetConsoleMode(h_Console, & dw_Mode))
384  {
385  GetTraces().SafeTrace(CLI_WIN_CONSOLE_OUT, *this) << "GetConsoleMode(line=" << __LINE__ << ") failed (" << (uint64_t) ::GetLastError() << ")" << endl;
386  break;
387  }
388 
389  // Succesful return.
390  return ScreenInfo(
391  t_ScreenBufferInfo.srWindow.Right - t_ScreenBufferInfo.srWindow.Left + 1,
392  t_ScreenBufferInfo.srWindow.Bottom - t_ScreenBufferInfo.srWindow.Top + 1,
393  true, // True Cls (if this line is reached, it means CleanScreen would succeed)
394  (dw_Mode & ENABLE_WRAP_AT_EOL_OUTPUT) != 0 // Line wrapping
395  );
396  } while(0);
397 
398  // Fallback to default behaviour.
399  return IODevice::GetScreenInfo();
