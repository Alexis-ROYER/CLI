98 Cli::~Cli(void)
99 {
100  GetTraces().Trace(TRACE_CLI) << "Deletion of CLI '" << GetName() << "'." << cli::endl;
101 
102  // Retrieve inner information before further deletions.
103  // !!! This method call is a little bit touchy since it is operated while the object is partially destructed at this point.
104  // However, IsConfigMenuEnabled() calls SyntaxNode features which part of this object has not been deleted yet.
105  const bool b_ConfigMenuEnabled = IsConfigMenuEnabled();
106 
107  // Remove this CLI from the global CLI list.
108  for ( Cli::List::Iterator it = GetCliRegistry().GetIterator();
109  GetCliRegistry().IsValid(it);
110  GetCliRegistry().MoveNext(it))
111  {
112  if (GetCliRegistry().GetAt(it) == this)
113  {
114  GetCliRegistry().Remove(it);
115  break;
116  }
117  }
118 
119  // Destroy menus.
120  while (! m_qMenus.IsEmpty())
121  {
122  if (const Menu* const pcli_Menu = m_qMenus.RemoveTail())
123  {
124  delete pcli_Menu;
125  }
126  }
127 
128  if (! b_ConfigMenuEnabled)
129  {
130  // When the config menu is disabled, m_pcliConfigMenuNode is not registered in SyntaxNode elements,
131  // thus it not freed by the SyntaxNode destructor.
132  delete m_pcliConfigMenuNode; // [contrib: Oleg Smolsky, 2010, based on CLI 2.5]
133  }
134 }

136 const int Cli::FindFromName(Cli::List& CLI_CliList, const char* const STR_RegExp)
137 {
138 #ifndef CLI_NO_REGEX
139  // Compile the regular expression.
140  regex_t s_RegExp;
141  memset(& s_RegExp, '\0', sizeof(regex_t));
142  if (regcomp(& s_RegExp, STR_RegExp, 0) != 0)
143  {
144  return -1;
145  }
146 #endif
147 
148  // Check the name for each CLI instance.
149  const int i_InitialCount = CLI_CliList.GetCount();
150  for ( Cli::List::Iterator it = GetCliRegistry().GetIterator();
151  GetCliRegistry().IsValid(it);
152  GetCliRegistry().MoveNext(it))
153  {
154  if (const Cli* const pcli_Cli = GetCliRegistry().GetAt(it))
155  {
156 #ifndef CLI_NO_REGEX
157  if (regexec(& s_RegExp, pcli_Cli->GetName(), 0, NULL, 0) == 0)
158 #endif
159  {
160  if (! CLI_CliList.AddTail(pcli_Cli))
161  {
162  CLI_ASSERT(false);
163  }
164  }
165  }
166  }
167 
168 #ifndef CLI_NO_REGEX
169  // Free.
170  regfree(& s_RegExp);
171 #endif
172 
173  return ((int) CLI_CliList.GetCount() - i_InitialCount);
174 }

192 const Menu* const Cli::GetMenu(const char* const STR_MenuName) const
193 {
194  // First of all, look in the menu list
195  for ( tk::Queue<const Menu*>::Iterator it = m_qMenus.GetIterator();
196  m_qMenus.IsValid(it);
197  m_qMenus.MoveNext(it))
198  {
199  if (const Menu* const pcli_Menu = m_qMenus.GetAt(it))
200  {
201  if (pcli_Menu->GetName() == STR_MenuName)
202  {
203  return pcli_Menu;
204  }
205  }
206  }
207 
208  // Possibly look for the CLI itself.
209  if (GetName() == STR_MenuName)
210  {
211  return this;
212  }
213 
214  return NULL;
215 }

296 const bool Cli::ExecuteReserved(const CommandLine& CLI_CommandLine) const
297 {
298  CommandLineIterator it(CLI_CommandLine);
299 
300  if (! it.StepIt()) { return false; }
301  else if (it == GetConfigMenuNode())
302  {
303  if (! it.StepIt()) { return false; }
304  if (dynamic_cast<const Endl*>(*it))
305  {
306  // Do nothing but return true so that an execution error is not detected.
307  return true;
308  }
309  }
310  #ifdef _DEBUG
311  else if (it == GetTracesMenuNode())
312  {
313  if (! it.StepIt()) { return false; }
314  if (dynamic_cast<const Endl*>(*it))
315  {
316  // Do nothing but return true so that an execution error is not detected.
317  return true;
318  }
319  }
320  #endif
321 
322  return Menu::ExecuteReserved(CLI_CommandLine);
323 }

353 const bool Cli::IsConfigMenuEnabled(void) const
354 {
355  CLI_ASSERT(m_pcliConfigMenuNode != NULL);
356 
357  Element::List cli_ExactList(10), cli_NearList(10);
358  if (FindElements(cli_ExactList, cli_NearList, m_pcliConfigMenuNode->GetKeyword()))
359  {
360  if (cli_ExactList.GetCount() == 1)
361  {
362  if (cli_ExactList.GetHead() == m_pcliConfigMenuNode)
363  {
364  return true;
365  }
366  }
367  }
368 
369  return false;
370 }
