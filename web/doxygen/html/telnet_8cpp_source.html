110 void TelnetServer::StartServer(void)
111 {
112  GetTraces().Trace(CLI_TELNET_SERVER) << "Starting server on port " << (uint32_t) m_ulTcpPort << endl;
113 
114  m_iServerSocket = socket(AF_INET, SOCK_STREAM, 0);
115 #ifdef CLI_WIN_NETWORK
116  // Winsock needs to be initialized.
117  if (m_iServerSocket < 0)
118  {
119  WORD wVersionRequested = MAKEWORD(2, 2);
120  WSADATA wsaData;
121  int err = WSAStartup(wVersionRequested, & wsaData);
122  if (err == 0)
123  {
124  ((int&) m_iServerSocket) = socket(AF_INET, SOCK_STREAM, 0);
125  }
126  }
127 #endif
128  if (m_iServerSocket < 0)
129  {
130  const ResourceString cli_Error = ResourceString()
131  .SetString(ResourceString::LANG_EN, "socket() failed")
132  .SetString(ResourceString::LANG_FR, "Echec de la fonction socket()");
133  OutputDevice::GetStdErr() << cli_Error.GetString(m_eLang) << endl;
134  GetTraces().Trace(CLI_TELNET_SERVER) << "errno = " << socket_errno << endl;
135  return;
136  }
137  GetTraces().Trace(CLI_TELNET_SERVER) << "socket() successful (m_iServerSocket = " << m_iServerSocket << ")" << endl;
138 
139  // So that we can re-bind to it without TIME_WAIT problems
140  int i_ReuseAddr = 1;
141  if (setsockopt(m_iServerSocket, SOL_SOCKET, SO_REUSEADDR, (const char*) & i_ReuseAddr, sizeof(i_ReuseAddr)) != 0)
142  {
143  close(m_iServerSocket);
144  const ResourceString cli_Error = ResourceString()
145  .SetString(ResourceString::LANG_EN, "setsockopt() failed")
146  .SetString(ResourceString::LANG_FR, "Echec de la fonction setsockopt()");
147  OutputDevice::GetStdErr() << cli_Error.GetString(m_eLang) << endl;
148  GetTraces().Trace(CLI_TELNET_SERVER) << "errno = " << socket_errno << endl;
149  return;
150  }
151 
152  // Bind the socket
153  struct sockaddr_in s_SockAddr;
154  s_SockAddr.sin_family = AF_INET;
155  s_SockAddr.sin_port = htons((unsigned short) m_ulTcpPort);
156  s_SockAddr.sin_addr.s_addr = INADDR_ANY;
157 
158  int i_BindTries = 0;
159  while (bind(m_iServerSocket, (struct sockaddr*) & s_SockAddr, sizeof(s_SockAddr)) < 0)
160  {
161  i_BindTries ++;
162 #ifndef CLI_WIN_NETWORK
163  if ((socket_errno == EADDRINUSE) && (i_BindTries < 2))
164 #else
165  if ((socket_errno == WSAEADDRINUSE) && (i_BindTries < 2))
166 #endif
167  {
168  sleep(500);
169  }
170  else
171  {
172  close(m_iServerSocket);
173  const ResourceString cli_Error = ResourceString()
174  .SetString(ResourceString::LANG_EN, "bind() failed")
175  .SetString(ResourceString::LANG_FR, "Echec de la fonction bind()");
176  OutputDevice::GetStdErr() << cli_Error.GetString(m_eLang) << endl;
177  GetTraces().Trace(CLI_TELNET_SERVER) << "errno = " << socket_errno << endl;
178  return;
179  }
180  }
181  GetTraces().Trace(CLI_TELNET_SERVER) << "bind(m_iServerSocket) successful" << endl;
182 
183  // Make the server listen up to m_uiMaxConnections clients.
184  if (listen(m_iServerSocket, m_uiMaxConnections) < 0)
185  {
186  close(m_iServerSocket);
187  const ResourceString cli_Error = ResourceString()
188  .SetString(ResourceString::LANG_EN, "listen() failed")
189  .SetString(ResourceString::LANG_FR, "Echec de la fonction listen()");
190  OutputDevice::GetStdErr() << cli_Error.GetString(m_eLang) << endl;
191  GetTraces().Trace(CLI_TELNET_SERVER) << "errno = " << socket_errno << endl;
192  return;
193  }
194  GetTraces().Trace(CLI_TELNET_SERVER) << "listen(m_iServerSocket) successful" << endl;
195 
196  GetTraces().Trace(CLI_TELNET_SERVER) << "Waiting for clients..." << endl;
197  while (m_iServerSocket >= 0)
198  {
199  if (! RunLoop(-1))
200  {
201  break;
202  }
203  }
204 
205  GetTraces().Trace(CLI_TELNET_SERVER) << "Server is turning off..." << endl;
206  TerminateServer();
207 
208  GetTraces().Trace(CLI_TELNET_SERVER) << "Server is done." << endl;
209 }

212 {
213  // First of all, unset the main socket reference to let the server loop know about the stop server procedure.
214  const int i_ServerSocket = m_iServerSocket;

225  OutputDevice::GetStdErr() << cli_Error.GetString(m_eLang) << endl;
226  GetTraces().Trace(CLI_TELNET_SERVER) << "errno = " << socket_errno << endl;
227  return;
228  }
229  GetTraces().Trace(CLI_TELNET_SERVER) << "close(m_iServerSocket) successful" << endl;
230  }

233 const bool TelnetServer::RunLoop(const int I_Milli)
234 {
235  // Prepare a fd_set structure to call select().
236  fd_set fd_Sockets; FD_ZERO(& fd_Sockets);
237  // Peek the server socket to detect new connections.
238  FD_SET(m_iServerSocket, & fd_Sockets);
239  // Peek known connections for input data and remember each known socket (for consistency).
240  tk::Queue<int> tk_Sockets(m_uiMaxConnections);
241  for ( tk::Map<int, ConnectionInfo>::Iterator it = m_tkConnections.GetIterator();
242  m_tkConnections.IsValid(it);
243  m_tkConnections.MoveNext(it))
244  {
245  const ConnectionInfo cli_Info = m_tkConnections.GetAt(it);
246  FD_SET(cli_Info.i_Socket, & fd_Sockets);
247  tk_Sockets.AddTail(cli_Info.i_Socket);
248  }
249 
250  // Set timeout as required.
251  timeval t_Timeout, *pt_Timeout = NULL;
252  if (I_Milli >= 0)
253  {
254  t_Timeout.tv_sec = I_Milli / 1000;
255  t_Timeout.tv_usec = (I_Milli * 1000) % 1000000;
256  pt_Timeout = & t_Timeout;
257  }
258 
259  // Check the server is still running.
260  if (m_iServerSocket < 0)
261  {
262  return false;
263  }
264 
265  // Use select.
266  GetTraces().Trace(CLI_TELNET_SERVER) << "calling select()..." << endl;
267  const int i_Select = select(FD_SETSIZE, & fd_Sockets, NULL, NULL, pt_Timeout);
268  GetTraces().Trace(CLI_TELNET_SERVER) << "select() returned " << i_Select << endl;
269  if (i_Select < 0)
270  {
271  GetTraces().Trace(CLI_TELNET_SERVER) << "select() failed, errno = " << socket_errno << endl;
272  return false;
273  }
274 
275  // Check the server is still running.
276  if (m_iServerSocket < 0)
277  {
278  return false;
279  }
280 
281  // Are there new connections?
282  if (FD_ISSET(m_iServerSocket, & fd_Sockets))
283  {
284  GetTraces().Trace(CLI_TELNET_SERVER) << "FD_ISSET(m_iServerSocket, & fd_Sockets) = " << FD_ISSET(m_iServerSocket, & fd_Sockets) << endl;
285  // New client.
286  if (! AcceptConnection())
287  {
288  // If the client cannot be accepted, let's continue whatever.
289  //return false;
290  }
291  }
292 
293  // Is there data for known connections?
294  while (! tk_Sockets.IsEmpty())
295  {
296  const int i_ConnectionSocket = tk_Sockets.RemoveHead();
297  if (const ConnectionInfo* const pcli_Info = m_tkConnections.GetAt(i_ConnectionSocket))
298  {
299  if (FD_ISSET(pcli_Info->i_Socket, & fd_Sockets))
300  {
301  if (TelnetConnection* const pcli_Connection = const_cast<TelnetConnection*>(pcli_Info->pcli_Connection))
302  {
303  GetTraces().Trace(CLI_TELNET_SERVER)
304  << "FD_ISSET(" << pcli_Info->i_Socket << " , & fd_Sockets) "
305  << "= " << FD_ISSET(pcli_Info->i_Socket, & fd_Sockets) << endl;
306 
307  bool b_Res = false;
308  if (pcli_Connection->ReceiveChars())
309  {
310  if (pcli_Connection->ProcessKeys())
311  {
312  b_Res = true;
313  }
314  }
315  if (! b_Res)
316  {
317  // [contrib: Dinesh Balasubramaniam, 2014, based on CLI 2.8]
318  // Client error / termination. Release socket resources so that pending/subsequent clients can be handled.
319  GetTraces().SafeTrace(CLI_TELNET_SERVER, *pcli_Connection) << "Error while receiving/processing keys. Closing telnet connection for socket " << i_ConnectionSocket << endl;
320  CloseConnection(i_ConnectionSocket);
321  }
322  }
323  }
324  }
325  }
326 
327  return true;
328 }

330 const bool TelnetServer::AcceptConnection(void)
331 {
332  // Accept the client
333  const int i_ConnectionSocket = accept(m_iServerSocket, NULL, NULL);
334  if (i_ConnectionSocket < 0)
335  {
336  const ResourceString cli_Error = ResourceString()
337  .SetString(ResourceString::LANG_EN, "accept() failed")
338  .SetString(ResourceString::LANG_FR, "Echec de la fonction accept()");
339  OutputDevice::GetStdErr() << cli_Error.GetString(m_eLang) << endl;
340  GetTraces().Trace(CLI_TELNET_SERVER) << "errno = " << socket_errno << endl;
341  return false;
342  }
343  GetTraces().Trace(CLI_TELNET_SERVER) << "accept() successful (i_ConnectionSocket = " << i_ConnectionSocket << ")" << endl;
344  const ResourceString cli_InfoClient = ResourceString()
345  .SetString(ResourceString::LANG_EN, "Accepting telnet connection")
346  .SetString(ResourceString::LANG_FR, "Connexion telnet acceptée");
347  GetTraces().Trace(CLI_TELNET_SERVER) << cli_InfoClient.GetString(m_eLang) << endl;
348 
349  // TelnetConnection / ExecutionContext configurations
350  if (TelnetConnection* const pcli_Connection = new TelnetConnection(this, i_ConnectionSocket, m_eLang, true))
351  {
352  // First check we do not have too many connections.
353  GetTraces().Trace(CLI_TELNET_SERVER) << "Already " << m_tkConnections.GetCount() << " active connections" << endl;
354  if (m_tkConnections.GetCount() >= m_uiMaxConnections)
355  {
356  const ResourceString cli_Error = ResourceString()
357  .SetString(ResourceString::LANG_EN, "Too many connections!")
358  .SetString(ResourceString::LANG_FR, "Nombre de connexions dépassé.");
359  GetTraces().Trace(CLI_TELNET_SERVER) << cli_Error.GetString(m_eLang) << endl;
360  if (pcli_Connection->OpenUp(__CALL_INFO__))
361  {
362  *pcli_Connection << cli_Error.GetString(m_eLang) << endl;
363  const ResourceString cli_Sorry = ResourceString()
364  .SetString(ResourceString::LANG_EN, "Too many connections! Sorry.")
365  .SetString(ResourceString::LANG_FR, "Nombre de connexions dépassé. Désolé.");
366  *pcli_Connection << cli_Sorry.GetString(m_eLang) << endl;
367  pcli_Connection->CloseDown(__CALL_INFO__);
368  }
369  }
370  else
371  {
372  // Require an execution context for the given connection.
373  if (ExecutionContext* const pcli_Context = OnNewConnection(*pcli_Connection))
374  {
375  // Store in the connection registry.
376  const ConnectionInfo s_ConnectionInfo = { i_ConnectionSocket, pcli_Connection, pcli_Context };
377  if ((! m_tkConnections.IsSet(i_ConnectionSocket))
378  && m_tkConnections.SetAt(i_ConnectionSocket, s_ConnectionInfo))
379  {
380  // Run the shell on that telnet connection.
381  pcli_Connection->UseInstance(__CALL_INFO__);
382  pcli_Context->Run(*pcli_Connection);
383 
384  // ExecutionContext automatically configures the stream as a trace stream.
385  // However in case of telnet connections, this may not be convenient by default.
386  GetTraces().UnsetStream(*pcli_Connection);
387 
388  // Since telnet connections are non-blocking devices, let's return successfully right now.
389  return true;
390  }
391 
392  // Fallback release.
393  OnCloseConnection(*pcli_Connection, pcli_Context);
394  }
395  }
396  // Fallback release.
397  delete pcli_Connection;
398  }
399  // Fallback release.
400  close(i_ConnectionSocket);
401  return false;
402 }

404 const bool TelnetServer::CloseConnection(const int I_ConnectionSocket)
405 {
406  GetTraces().Trace(CLI_TELNET_SERVER) << "Ending connection " << I_ConnectionSocket << "..." << endl;
407 
408  // Release objects.
409  if (const ConnectionInfo* const pcli_Info = m_tkConnections.GetAt(I_ConnectionSocket))
410  {
411  if (TelnetConnection* const pcli_Connection = pcli_Info->pcli_Connection)
412  {
413  if (ExecutionContext* const pcli_Context = pcli_Info->pcli_Context)
414  {
415  OnCloseConnection(*pcli_Connection, pcli_Context);
416  }
417  pcli_Connection->FreeInstance(__CALL_INFO__);
418  }
419  m_tkConnections.Unset(I_ConnectionSocket);
420  }
421 
422  return true;
423 }

425 const bool TelnetServer::TerminateServer(void)
426 {
427  bool b_Res = true;
428 
429  // Close all client connections.
430  while (! m_tkConnections.IsEmpty())
431  {
432  tk::Map<int, ConnectionInfo>::Iterator it = m_tkConnections.GetIterator();
433  if (m_tkConnections.IsValid(it))
434  {
435  const ConnectionInfo cli_Info = m_tkConnections.Remove(it);
436 
437  // Let the shell terminate.
438  if (cli_Info.pcli_Context != NULL)
439  {
440  if (cli_Info.pcli_Context->IsRunning())
441  {
442  GetTraces().Trace(CLI_TELNET_SERVER) << "Terminating shell of connection " << cli_Info.i_Socket << "..." << cli::endl;
443  cli_Info.pcli_Context->StopExecution();
444  }
445  }
446  }
447  }
448 
449  return b_Res;
450 }

468 const bool TelnetConnection::OpenDevice(void)
469 {
470  // Configure LINGER option
471  linger t_Linger;
472  t_Linger.l_onoff = 1; // enable LINGER
473  t_Linger.l_linger = 1; // seconds
474  if (setsockopt(m_iSocket, SOL_SOCKET, SO_LINGER, (const char*) & t_Linger, sizeof(t_Linger)) != 0)
475  {
476  const ResourceString cli_Error = ResourceString()
477  .SetString(ResourceString::LANG_EN, "setsockopt(SO_LINGER) failed")

479  OutputDevice::GetStdErr() << cli_Error.GetString(m_eLang) << endl;
480  GetTraces().SafeTrace(CLI_TELNET_SERVER, *this) << "errno = " << socket_errno << endl;
481  CloseDevice();
482  return false;
483  }
484  GetTraces().SafeTrace(CLI_TELNET_SERVER, *this) << "setsockopt(" << m_iSocket << ", SO_LINGER) successful" << endl;
485 
486  return true;

489 const bool TelnetConnection::CloseDevice(void)
490 {
491  // Did not manage to correctly end up the connection on the server side,
492  // neither on Windows nor Linux systems.
493  // This a hack until I provide another implementation.
494  {
495  linger t_Linger; socklen_t i_Len = sizeof(t_Linger);
496  if (getsockopt(m_iSocket, SOL_SOCKET, SO_LINGER, (char*) & t_Linger, & i_Len) != 0)
497  {
498  close(m_iSocket);
499  const ResourceString cli_Error = ResourceString()
500  .SetString(ResourceString::LANG_EN, "getsockopt(SO_LINGER) failed")
501  .SetString(ResourceString::LANG_FR, "Echec de la fonction getsockopt(SO_LINGER)");
502  OutputDevice::GetStdErr() << cli_Error.GetString(m_eLang) << endl;
503  GetTraces().SafeTrace(CLI_TELNET_SERVER, *this) << "errno = " << socket_errno << endl;
504  return false;
505  }
506  GetTraces().SafeTrace(CLI_TELNET_SERVER, *this) << "getsockopt(" << m_iSocket << ", SO_LINGER): onoff = " << t_Linger.l_onoff << endl;
507  GetTraces().SafeTrace(CLI_TELNET_SERVER, *this) << "getsockopt(" << m_iSocket << ", SO_LINGER): linger = " << t_Linger.l_linger << endl;
508 
509  if (t_Linger.l_onoff)
510  {
511  sleep(t_Linger.l_linger);
512  }
513  }
514 
515  // Stop sending characters
516  if (shutdown(m_iSocket, SHUT_WR) != 0)
517  {
518  close(m_iSocket);
519  const ResourceString cli_Error = ResourceString()
520  .SetString(ResourceString::LANG_EN, "shutdown() failed")
521  .SetString(ResourceString::LANG_FR, "Echec de la fonction shutdown()");
522  OutputDevice::GetStdErr() << cli_Error.GetString(m_eLang) << endl;
523  GetTraces().SafeTrace(CLI_TELNET_SERVER, *this) << "errno = " << socket_errno << endl;
524  return false;
525  }
526  GetTraces().SafeTrace(CLI_TELNET_SERVER, *this) << "shutdown(" << m_iSocket << ") successful" << endl;
527 
528  // Release resources
529  if (close(m_iSocket) != 0)
530  {
531  const ResourceString cli_Error = ResourceString()
532  .SetString(ResourceString::LANG_EN, "close() failed")
533  .SetString(ResourceString::LANG_FR, "Echec de la fonction close()");
534  OutputDevice::GetStdErr() << cli_Error.GetString(m_eLang) << endl;
535  GetTraces().SafeTrace(CLI_TELNET_SERVER, *this) << "errno = " << socket_errno << endl;
536  return false;
537  }
538  GetTraces().SafeTrace(CLI_TELNET_SERVER, *this) << "close(" << m_iSocket << ") successful" << endl;
539 
540  return true;

543 const bool TelnetConnection::ReceiveChars(void) const
544 {
545  // Dequeue characters from the socket
546  GetTraces().SafeTrace(CLI_TELNET_IN, *this) << "m_iSocket = " << m_iSocket << ", calling recv()..." << endl;
547  char arc_Buffer[256];
548  const int i_LenMax = ((CLI_TELNET_INPUT_BUFFER_SIZE - m_qChars.GetCount() < 256) ? (CLI_TELNET_INPUT_BUFFER_SIZE - m_qChars.GetCount()) : 256);
549  const int i_Len = recv(m_iSocket, arc_Buffer, i_LenMax, 0);
550  GetTraces().SafeTrace(CLI_TELNET_IN, *this) << "recv() returned " << i_Len << endl;
551  if (i_Len <= 0)
552  {
553  m_cliLastError
554  .SetString(ResourceString::LANG_EN, "Telnet reception error")
555  .SetString(ResourceString::LANG_FR, "Echec de réception en telnet");
556  GetTraces().SafeTrace(CLI_TELNET_IN, *this) << "recv() failed, errno = " << socket_errno << endl;
557  return false; // [contrib: Dinesh Balasubramaniam, 2014, based on CLI 2.8]
558  }
559 
560  // Queue them in the internal fifo
561  for (int i=0; i<i_Len; i++)
562  {
563  if (arc_Buffer[i] != '\0')
564  {
565  if (! m_qChars.AddTail(arc_Buffer[i]))
566  {
