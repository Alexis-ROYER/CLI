52 ResourceString& ResourceString::operator=(const ResourceString& CLI_String)
53 {
54  for ( LANG e_Lang = (LANG) 0;
55  e_Lang < LANG_COUNT;
56  e_Lang = (LANG) (((int) e_Lang) + 1))
57  {
58  if (CLI_String.HasString(e_Lang))
59  {
60  SetString(e_Lang, CLI_String.GetString(e_Lang));
61  }
62  else
63  {
64  SetString(e_Lang, "");
65  }
66  }
67  return *this;
68 }

107 const bool ResourceString::IsEmpty(void) const
108 {
109  for (int e_Lang = 0; e_Lang < LANG_COUNT; e_Lang ++)
110  {
111  if (! GetString((LANG) e_Lang).IsEmpty())
112  {
113  // If one language has a non empty string set, then the resource string is not empty.
114  return false;
115  }
116  }
117 
118  // If no language has a string set, then the resource string is empty.
119  return true;
120 }

152 const ResourceString cli::operator+(const ResourceString& CLI_Str1, const ResourceString& CLI_Str2)
153 {
154  ResourceString cli_Result;
155  for (int e_Lang = 0; e_Lang < ResourceString::LANG_COUNT; e_Lang ++)
156  {
157  // Concatenation for each language.
158  cli_Result.SetString((ResourceString::LANG) e_Lang,
159  ResourceString::Concat(
160  CLI_Str1.GetString((ResourceString::LANG) e_Lang),
161  CLI_Str2.GetString((ResourceString::LANG) e_Lang)));
162  }
163  return cli_Result;
164 }
