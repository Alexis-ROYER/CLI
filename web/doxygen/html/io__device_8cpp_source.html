161 const bool OutputDevice::CloseDown(const CallInfo& CLI_CallInfo)
162 {
163  bool b_Res = true;
164 
165  if (m_iOpenLock > 0)
166  {
167  GetTraces().SafeTrace(TRACE_IO_DEVICE_OPENING, *this)
168  << "One less user for instance " << GetDebugName() << ", "
169  << "user count: " << m_iOpenLock << " -> " << m_iOpenLock - 1 << ", "
170  << "from " << CLI_CallInfo.GetFunction() << " "
171  << "at " << CLI_CallInfo.GetFile() << ":" << CLI_CallInfo.GetLine() << endl;
172 
173  if (m_iOpenLock == 1)
174  {
175  GetTraces().SafeTrace(TRACE_IO_DEVICE_OPENING, *this)
176  << "Closing the device " << GetDebugName() << endl;
177  b_Res = CloseDevice();
178  }
179 
180  m_iOpenLock --;
181  }
182  else
183  {
184  GetTraces().SafeTrace(TRACE_IO_DEVICE_OPENING, *this)
185  << "No more closing down for instance " << GetDebugName() << ", "
186  << "user count = " << m_iOpenLock << ", "
187  << "from " << CLI_CallInfo.GetFunction() << " "
188  << "at " << CLI_CallInfo.GetFile() << ":" << CLI_CallInfo.GetLine() << endl;
189  }
190 
191  return b_Res;
192 }

284 const OutputDevice& OutputDevice::operator <<(const double D_Out) const
285 {
286  double d_Out = D_Out;
287 
288  // Positive / negative number management
289  const char* str_Sign = "";
290  if (d_Out < 0.0)
291  {
292  str_Sign = "-";
293  d_Out = - d_Out;
294  }
295 
296  // Reframe very big or very small numbers so that it can be displayed with an appropriate number of digits.
297  int i_Exp = 0;
298  if (d_Out > 1e6)
299  {
300  while (d_Out >= 1000.0)
301  {
302  d_Out /= 1000.0;
303  i_Exp += 3;
304  }
305  }
306  if ((d_Out < 1e-6) && (d_Out != 0.0))
307  {
308  while (d_Out < 1.0)
309  {
310  d_Out *= 1000.0;
311  i_Exp -= 3;
312  }
313  }
314 
315  // Let's print out 6 digits max.
316  // Avoid useless trailing '0' in the decimal (after the coma).
317  // double d_Decimals = d_Out;
318  // d_tmp -= (int) d_tmp;
319  int i_Decimal, i_Out;
320  for ( i_Decimal = 6, i_Out = (int) floor(d_Out * 1e6 + 0.5); // floor(D+1/2) should be equivalent to round(D)
321  i_Decimal > 1;
322  i_Decimal --, i_Out /= 10)
323  {
324  if ((i_Out % 10) != 0)
325  {
326  break;
327  }
328  }
329 
330  // Format the float number with the computed characteristics.
331  char str_Format[128], str_Out[128];
332  if (i_Exp == 0)
333  {
334  snprintf(str_Format, sizeof(str_Format), "%%s%%.%df", i_Decimal); // No call needed to CheckSnprintfResult() here.
335  const int i_Res = snprintf(str_Out, sizeof(str_Out), str_Format, str_Sign, d_Out);
336  CheckSnprintfResult(str_Out, sizeof(str_Out), i_Res);
337  }
338  else
339  {
340  snprintf(str_Format, sizeof(str_Format), "%%s%%.%dfe%%d", i_Decimal); // No call needed to CheckSnprintfResult() here.
341  const int i_Res = snprintf(str_Out, sizeof(str_Out), str_Format, str_Sign, d_Out, i_Exp);
342  CheckSnprintfResult(str_Out, sizeof(str_Out), i_Res);
343  }
344 
345  // Output the computed string.
346  PutString(str_Out);
347 
348  // Return the instance itself.
349  return *this;
350 }
